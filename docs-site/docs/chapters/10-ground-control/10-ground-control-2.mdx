# Ground Control — Part 2

## 6. Android (Kotlin) Implementation

The Android side uses `BiometricPrompt` from the AndroidX biometric library.

### Step 1: Add the dependency

```groovy title="android/app/build.gradle"
dependencies {
    implementation "androidx.biometric:biometric:1.1.0"
}
```

### Step 2: Create the plugin class

```kotlin title="android/app/src/main/kotlin/.../BiometricPlugin.kt"
package com.corebank.app

import android.content.Context
import androidx.biometric.BiometricManager
import androidx.biometric.BiometricPrompt
import androidx.core.content.ContextCompat
import androidx.fragment.app.FragmentActivity
import io.flutter.embedding.engine.plugins.FlutterPlugin
import io.flutter.embedding.engine.plugins.activity.ActivityAware
import io.flutter.embedding.engine.plugins.activity.ActivityPluginBinding
import io.flutter.plugin.common.MethodCall
import io.flutter.plugin.common.MethodChannel
import io.flutter.plugin.common.MethodChannel.MethodCallHandler

class BiometricPlugin : FlutterPlugin, MethodCallHandler, ActivityAware {

    private lateinit var channel: MethodChannel
    private var activity: FragmentActivity? = null

    override fun onAttachedToEngine(binding: FlutterPlugin.FlutterPluginBinding) {
        channel = MethodChannel(binding.binaryMessenger, "com.corebank/biometrics")
        channel.setMethodCallHandler(this)
    }

    override fun onDetachedFromEngine(binding: FlutterPlugin.FlutterPluginBinding) {
        channel.setMethodCallHandler(null)
    }

    override fun onAttachedToActivity(binding: ActivityPluginBinding) {
        activity = binding.activity as? FragmentActivity
    }

    override fun onDetachedFromActivity() { activity = null }
    override fun onReattachedToActivityForConfigChanges(binding: ActivityPluginBinding) {
        activity = binding.activity as? FragmentActivity
    }
    override fun onDetachedFromActivityForConfigChanges() { activity = null }

    override fun onMethodCall(call: MethodCall, result: MethodChannel.Result) {
        when (call.method) {
            "isAvailable" -> handleIsAvailable(result)
            "authenticate" -> {
                val reason = call.argument<String>("reason") ?: "Authenticate"
                handleAuthenticate(reason, result)
            }
            else -> result.notImplemented()
        }
    }

    private fun handleIsAvailable(result: MethodChannel.Result) {
        val context = activity ?: run {
            result.success(false)
            return
        }
        val manager = BiometricManager.from(context)
        val canAuth = manager.canAuthenticate(
            BiometricManager.Authenticators.BIOMETRIC_STRONG
        )
        result.success(canAuth == BiometricManager.BIOMETRIC_SUCCESS)
    }

    private fun handleAuthenticate(reason: String, result: MethodChannel.Result) {
        val fragmentActivity = activity ?: run {
            result.error("NO_ACTIVITY", "No activity available", null)
            return
        }

        val executor = ContextCompat.getMainExecutor(fragmentActivity)

        val callback = object : BiometricPrompt.AuthenticationCallback() {
            override fun onAuthenticationSucceeded(authResult: BiometricPrompt.AuthenticationResult) {
                result.success(true)
            }

            override fun onAuthenticationFailed() {
                result.success(false)
            }

            override fun onAuthenticationError(errorCode: Int, errString: CharSequence) {
                if (errorCode == BiometricPrompt.ERROR_USER_CANCELED ||
                    errorCode == BiometricPrompt.ERROR_NEGATIVE_BUTTON) {
                    result.error("USER_CANCEL", errString.toString(), null)
                } else {
                    result.error("AUTH_FAILED", errString.toString(), null)
                }
            }
        }

        val prompt = BiometricPrompt(fragmentActivity, executor, callback)

        val info = BiometricPrompt.PromptInfo.Builder()
            .setTitle("CoreBank")
            .setSubtitle(reason)
            .setNegativeButtonText("Cancel")
            .build()

        prompt.authenticate(info)
    }
}
```

### Step 3: Register in MainActivity

```kotlin title="android/app/src/main/kotlin/.../MainActivity.kt"
package com.corebank.app

import io.flutter.embedding.android.FlutterFragmentActivity
import io.flutter.embedding.engine.FlutterEngine

class MainActivity : FlutterFragmentActivity() {
    override fun configureFlutterEngine(flutterEngine: FlutterEngine) {
        super.configureFlutterEngine(flutterEngine)
        flutterEngine.plugins.add(BiometricPlugin())
    }
}
```

Note: `FlutterFragmentActivity` instead of `FlutterActivity` — required for `BiometricPrompt`.

## 7. Conditional Platform Code

Not every platform supports every feature. CoreBank runs on iOS, Android, and web — and web has no biometric API.

```dart title="lib/utils/platform_utils.dart"
import 'dart:io' show Platform;
import 'package:flutter/foundation.dart' show kIsWeb;

class PlatformUtils {
  /// True if running in a web browser.
  static bool get isWeb => kIsWeb;

  /// True if running on iOS (not web).
  static bool get isIOS => !kIsWeb && Platform.isIOS;

  /// True if running on Android (not web).
  static bool get isAndroid => !kIsWeb && Platform.isAndroid;

  /// True if biometric auth could potentially be available.
  static bool get supportsBiometrics => isIOS || isAndroid;
}
```

:::tip[WHY THIS MATTERS]
You must check `kIsWeb` before accessing `Platform.isIOS` or `Platform.isAndroid`. The `dart:io` `Platform` class does not exist on web — importing it is fine, but calling its properties will throw. Always gate with `kIsWeb` first.

:::

Use it in the login screen:

```dart
if (PlatformUtils.supportsBiometrics && _biometricsAvailable) ...[
  OutlinedButton.icon(
    onPressed: _authenticateWithBiometrics,
    icon: const Icon(Icons.fingerprint),
    label: const Text('Use Biometrics'),
  ),
],
```

For more nuanced platform checks, use `defaultTargetPlatform`:

```dart
switch (defaultTargetPlatform) {
  case TargetPlatform.iOS:
    // iOS-specific behavior
  case TargetPlatform.android:
    // Android-specific behavior
  default:
    // Fallback for web, desktop, etc.
}
```

## 8. Using Existing Plugins — local_auth

You just built biometric auth from scratch to understand the mechanism. In production, you would use the [`local_auth`](https://pub.dev/packages/local_auth) package, which wraps exactly what we built — and more.

```yaml title="pubspec.yaml"
dependencies:
  local_auth: ^2.2.0
```

```dart title="lib/services/biometric_service.dart"
import 'package:local_auth/local_auth.dart';

class BiometricService {
  final LocalAuthentication _auth = LocalAuthentication();

  Future<bool> isAvailable() async {
    final canCheck = await _auth.canCheckBiometrics;
    final isDeviceSupported = await _auth.isDeviceSupported();
    return canCheck && isDeviceSupported;
  }

  Future<bool> authenticate() async {
    return _auth.authenticate(
      localizedReason: 'Authenticate to access CoreBank',
      options: const AuthenticationOptions(
        stickyAuth: true,
        biometricOnly: true,
      ),
    );
  }
}
```

Compare the two approaches:

| Aspect | Manual MethodChannel | local_auth Package |
|---|---|---|
| Lines of Dart code | ~40 | ~15 |
| Lines of native code | ~150 (Swift + Kotlin) | 0 |
| Error handling | Manual per platform | Built-in |
| Platform support | iOS + Android | iOS + Android + Windows + macOS |
| Maintenance | You own it | Community maintained |

:::info[TRY IT YOURSELF]
Replace the manual `BiometricChannel` with `local_auth` in your CoreBank project. The Dart-side API is nearly identical. This exercise shows how plugins are simply convenience wrappers around the MethodChannel patterns you just learned.

:::

## 9. Writing a Simple Plugin Package

If you find yourself copying platform channel code between projects, package it as a plugin.

```
my_biometric_plugin/
  lib/
    my_biometric_plugin.dart          # Dart API
    my_biometric_plugin_platform_interface.dart  # Abstract platform interface
    my_biometric_plugin_method_channel.dart      # MethodChannel implementation
  android/
    src/main/kotlin/.../MyBiometricPlugin.kt
  ios/
    Classes/MyBiometricPlugin.swift
  pubspec.yaml
  README.md
```

Create the scaffold:

```bash
flutter create --template=plugin --platforms=android,ios my_biometric_plugin
```

This generates the full structure with:
- A Dart API layer
- A platform interface (for testability and federation)
- A MethodChannel implementation
- Native code skeletons for each platform
- Example app for testing

The platform interface pattern is important: it lets you swap the implementation for testing (mock) or for different platforms (web uses a different strategy than mobile).

```dart title="lib/my_biometric_plugin_platform_interface.dart"
abstract class MyBiometricPluginPlatform {
  Future<bool> isAvailable();
  Future<bool> authenticate({required String reason});
}
```

```dart title="lib/my_biometric_plugin_method_channel.dart"
class MethodChannelMyBiometricPlugin extends MyBiometricPluginPlatform {
  final _channel = const MethodChannel('com.example/my_biometric_plugin');

  @override
  Future<bool> isAvailable() async {
    return await _channel.invokeMethod<bool>('isAvailable') ?? false;
  }

  @override
  Future<bool> authenticate({required String reason}) async {
    return await _channel.invokeMethod<bool>(
      'authenticate',
      {'reason': reason},
    ) ?? false;
  }
}
```

class LoginScreen extends StatelessWidget {
  @override

    return Column(

Every login requires typing a password. Slow, frustrating, and less secure (users pick weak passwords to type quickly).

<TabItem value="after" label="After">

```dart title="Biometric + password login"
class LoginScreen extends ConsumerStatefulWidget { /* ... */ }

class _LoginScreenState extends ConsumerState<LoginScreen> {
  bool _biometricsAvailable = false;

  @override
  void initState() {
    super.initState();
    if (PlatformUtils.supportsBiometrics) {
      _checkBiometrics();
    }
  }

  Future<void> _checkBiometrics() async {
    final available = await BiometricChannel.isAvailable();
    if (mounted) setState(() => _biometricsAvailable = available);
  }

  @override
  Widget build(BuildContext context) {
    return Column(
      children: [
        // Email and password fields (Form with validation)
        TextFormField(/* email */),
        TextFormField(/* password */),
        FilledButton(
          onPressed: _loginWithPassword,
          child: const Text('Log In'),
        ),
        // Biometric option — only shown when available
        if (_biometricsAvailable) ...[
          const Padding(
            padding: EdgeInsets.symmetric(vertical: 12),
            child: Row(
              children: [
                Expanded(child: Divider()),
                Padding(
                  padding: EdgeInsets.symmetric(horizontal: 8),
                  child: Text('or'),
                ),
                Expanded(child: Divider()),
              ],
            ),
          ),
          OutlinedButton.icon(
            onPressed: _authenticateWithBiometrics,
            icon: Icon(
              PlatformUtils.isIOS
                  ? Icons.face
                  : Icons.fingerprint,
            ),
            label: Text(
              PlatformUtils.isIOS
                  ? 'Log in with Face ID'
                  : 'Log in with Fingerprint',
            ),
          ),
        ],
      ],
    );
  }
}
```

Users tap once for biometric login. Password remains as fallback. The button label and icon adapt per platform. Web users see only the password form.
